代码如下： //注意left和right结束的判别条件 先后
int search(int* nums, int numsSize, int target){

int left=0;

int right=numsSize-1;

int mid=(right-left)/2+left;

while(left<=right)

{

if(nums[mid]==target){

return mid;

}

else if(nums[mid]>target){

right=mid-1;

mid=(right-left)/2+left;

}

else{

left=mid +1;

mid=(left+right)/2;

}

}

return -1;

}
